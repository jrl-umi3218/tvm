199 template<class TD, class TDImpl>
200 inline void Clamped<TD, TDImpl>::Impl::updateValue()
201 {
202  TDImpl::updateValue();
203  double s = 1;
204  for(int i = 0; i < this->function().size(); ++i)
205  {
206  if(this->value_[i] > max_[i])
207  {
208  // innerValue[i] > max_[i] >= 0 so that innerValue[i] != 0
209  s = std::min(s, max_[i] / this->value_[i]);
210  }
211  else if(this->value_[i] < min_[i])
212  {
213  // this->value_[i] < min_[i] <= 0 so that this->value_[i] != 0
214  s = std::min(s, min_[i] / this->value_[i]);
215  }
216  }
217  this->value_ *= s;
218 }

100 template<typename CstDerived, typename... Derived>
101 BasicLinearFunction::BasicLinearFunction(const utils::AffineExpr<CstDerived, Derived...> & aff)
102 : LinearFunction(static_cast<int>(std::get<0>(aff.linear()).matrix().rows()))
103 {
104  using Indices = std::make_index_sequence<sizeof...(Derived)>;
105  tvm::internal::VariableCountingVector v;
106  internal::addVar(v, aff.linear(), Indices{});
107  const auto & vars = v.variables();
108  const auto & simple = v.simple(); // simple[i]==true means vars[i] that appears in v based on a single first add to v
109  for(int i = 0; i < vars.numberOfVariables(); ++i)
110  {
111  if(!simple[i]) // if vars[i] is the results of adding several potentially overlapping variables to v...
112  {
113  addVariable(vars[i], true); // ... we add the variable now and initialize its jacobian to 0 below.
114  // This will force using += to fill the part of the jacobian corresponding to
115  // a subvariable in method add. Simple variables will be added directly in add.
116  }
117  }
118  for(auto & J : jacobian_)
119  {
120  J.second.setZero();
121  J.second.properties({tvm::internal::MatrixProperties::Constness(true)});
122  }
123  add(aff.linear(), Indices{});
124  if constexpr(std::is_same_v<CstDerived, utils::internal::NoConstant>)
125  {
126  this->b(Eigen::VectorXd::Zero(size()),
127  {tvm::internal::MatrixProperties::Constness(true), tvm::internal::MatrixProperties::ZERO});
128  }
129  else
130  {
131  this->b(aff.constant(), {tvm::internal::MatrixProperties::Constness(true)});
132  }
133  setDerivativesToZero();
134 }

136 template<typename Derived>
137 inline void BasicLinearFunction::add(const Eigen::MatrixBase<Derived> & A, VariablePtr x)
138 {
139  if(!x->space().isEuclidean() && x->isBasePrimitive())
140  throw std::runtime_error("We allow linear function only on Euclidean variables.");
141  if(A.rows() != size())
142  throw std::runtime_error("Matrix A doesn't have coherent row size.");
143  if(A.cols() != x->size())
144  throw std::runtime_error("Matrix A doesn't have its column size coherent with its corresponding variable.");
145 
146  if(variables().contains(*x))
147  {
148  auto J = jacobian_.at(x.get(), tvm::utils::internal::with_sub{});
149  J += A; // this erase the properties of J;
150  J.properties({tvm::internal::MatrixProperties::Constness(true)}); // We restore only this one
151  }
152  else
153  {
154  using Shape = tvm::internal::MatrixProperties::Shape;
155  addVariable(x, true);
156  jacobian_.at(x.get()) = A;
157  Shape shape = Shape::GENERAL;
158  // TODO: generalize test to matrices with fixed size, to triangular matrices
159  // and to richer diagonal case
160  if constexpr(std::is_same_v<Derived, std::remove_const_t<tvm::utils::internal::IdentityType>>)
161  {
162  shape = Shape::IDENTITY;
163  }
164  if constexpr(std::is_same_v<Derived, std::remove_const_t<tvm::utils::internal::MinusIdentityType>>)
165  {
166  shape = Shape::MINUS_IDENTITY;
167  }
168  if constexpr(std::is_same_v<Derived, std::remove_const_t<tvm::utils::internal::MultIdentityType>>)
169  {
170  shape = Shape::MULTIPLE_OF_IDENTITY;
171  }
172  if constexpr(std::is_same_v<Derived, std::remove_const_t<decltype(Eigen::VectorXd(1).asDiagonal())>>)
173  {
174  shape = Shape::DIAGONAL;
175  }
176  jacobian_.at(x.get()).properties({tvm::internal::MatrixProperties::Constness(true), shape});
177  }
178 }
